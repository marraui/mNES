# cmake version to be used
cmake_minimum_required(VERSION 3.1)

set (CMAKE_CXX_STANDARD 11)

# project name
project(NES)

# include
include_directories(./include ./lib/SLD2/include)

# target
add_executable(NES ./src/main.cpp ./src/bus.cpp ./src/cpu6502.cpp)

# external libraries
add_subdirectory(./lib/SDL2)
target_link_libraries(NES SDL2)
